Validando a data

Instalar o date-fns

yarn add date-fns

parseISO, converte uma string para o formato date do JavaScript.

startOfHour, vai deixar a hora, minutos, segundos e milissegundos zerados.

IsEqual, verifica se duas datas sÃ£o iguais, na mesma data e no mesmo horÃ¡rio.

appointments.routes.ts

import { Router } from 'express';
import { uuid } from 'uuidv4';
import { startOfHour, parseISO, isEqual } from 'date-fns';

const appointmentsRouter = Router();

interface Appointment {
  id: string;
  provider: string;
  date: Date;
}

const appointments: Appointment[] = [];

appointmentsRouter.post('/', (request, response) => {
  const { provider, date } = request.body;

  // Tranformar a data no tipo date do Javascript
  const parsedDate = startOfHour(parseISO(date)); // "2020-04-14T16:00:00.000Z"

  // Verificar se existe uma data ja agendada.
  const findAppointmentInSameDate = appointments.find(appointment =>
    isEqual(parsedDate, appointment.date),
  );

	// Se existir, vai dar errro
  if (findAppointmentInSameDate) {
    return response
      .status(400)
      .json({ message: 'This appointment is already booked' });
  }

  const appointment = {
    id: uuid(),
    provider,
    date: parsedDate,
  };

  appointments.push(appointment);

  return response.json(appointment);
});

export default appointmentsRouter;
